Example 1: Example API Call

This example shows how to call FREME NER.

Example in cURL

curl -X POST -H "Content-Type: text/plain" -H "Accept: text/turtle" -H "Cache-Control: no-cache" -d 'Welcome to the city of Prague.' "http://api.freme-project.eu/0.5/e-entity/freme-ner/documents/?language=en&dataset=dbpedia"

<?php

$request = new HttpRequest();
$request->setUrl('http://api.freme-project.eu/0.5/e-entity/freme-ner/documents/');
$request->setMethod(HTTP_METH_POST);

$request->setQueryData(array(
  'language' => 'en',
  'dataset' => 'dbpedia'
));

$request->setHeaders(array(
  'cache-control' => 'no-cache',
  'accept' => 'text/turtle',
  'content-type' => 'text/plain'
));

$request->setBody('Welcome to the city of Prague.');

try {
  $response = $request->send();

  echo $response->getBody();
} catch (HttpException $ex) {
  echo $ex;
}
